    def compute_preference(self, value:float) -> float:
        """Compute the preference value

        Parameters
        ----------
        value : float
            the difference between the evaluations of the alternatives i and j for the criterion that use this preference function

        Returns
        -------
        float
            the value of preference between alternatives i and j
        """
        preference = 0.0
        type = self.type.get()
        pc = self.pc.get()
        qc = self.qc.get()

        if type == 1: # Usual
            if value > 0:
                preference = 1.0
        elif type == 2: # U-shape
            if value >= qc:
                preference = 1.0
        elif type == 3: # V-shape
            if value >= pc :
                preference = 1.0
            elif value > 0:
                preference = value/pc
        elif type == 4: # Level
            if value >= pc:
                preference = 1.0
            elif value >= qc:
                preference = 0.5
        elif type == 5: # Linear
            if value >= pc:
                preference = 1.0
            elif value > qc:
                preference = (value - qc)/(pc - qc)
        elif type == 6: # Gaussian
            if value > 0:
                preference = np.exp(-value**2)
        return preference
